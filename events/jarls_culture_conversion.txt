# The indo_norse Melting Pot
jarls_conversion.2 = {
	type = character_event
	desc = "EVTDESC_jarls_conversion.2"
	picture = GFX_evt_viking_throneroom_oldgods
	
	theme = culture_change

	trigger = {
		any_province = {
			OR = {
				region = world_india_bengal
				region = world_india_rajastan
				kingdom = { title = k_lanka }
			}
			has_culture_group = culture_group:indo_aryan_group
			NOT = { has_culture = culture:indo_norse }
		}
		NOT = { government = chinese_imperial_government }
	}

	option = {
		name = "EVTOPTA_jarls_conversion.2"
		tooltip_info = learning
		trigger = {
			OR = {
				trait = detached_priest
				trait = martial_cleric
				trait = scholarly_theologian
				trait = mastermind_theologian
			}
		}
		set_culture = culture:indo_norse
		random_demesne_province = {
			limit = {
				OR = {
					region = world_india_bengal
					region = world_india_rajastan
					kingdom = { title = k_lanka }
				}
				has_culture_group = culture_group:indo_aryan_group
				NOT = { has_culture = culture:indo_norse }
			}
			set_culture = culture:indo_norse
		}
	}
	option = {
		name = "EVTOPTB_jarls_conversion.2"
		tooltip_info = diplomacy
		trigger = {
			OR = {
				trait = naive_appeaser
				trait = underhanded_rogue
				trait = charismatic_negotiator
				trait = grey_eminence
			}
		}
		set_culture = culture:indo_norse
		random_demesne_province = {
			limit = {
				OR = {
					region = world_india_bengal
					region = world_india_rajastan
					kingdom = { title = k_lanka }
				}
				has_culture_group = culture_group:indo_aryan_group
				NOT = { has_culture = culture:indo_norse }
			}
			set_culture = culture:indo_norse
		}
	}
	option = {
		name = "EVTOPTC_jarls_conversion.2"
		tooltip_info = martial
		trigger = {
			OR = {
				trait = misguided_warrior
				trait = tough_soldier
				trait = skilled_tactician
				trait = brilliant_strategist
			}
		}
		set_culture = culture:indo_norse
		random_demesne_province = {
			limit = {
				OR = {
					region = world_india_bengal
					region = world_india_rajastan
					kingdom = { title = k_lanka }
				}
				has_culture_group = culture_group:indo_aryan_group
				NOT = { has_culture = culture:indo_norse }
			}
			set_culture = culture:indo_norse
		}
	}
	option = {
		name = "EVTOPTD_jarls_conversion.2"
		tooltip_info = stewardship
		trigger = {
			OR = {
				trait = indulgent_wastrel
				trait = thrifty_clerk
				trait = fortune_builder
				trait = midas_touched
			}
		}
		set_culture = culture:indo_norse
		random_demesne_province = {
			limit = {
				OR = {
					region = world_india_bengal
					region = world_india_rajastan
					kingdom = { title = k_lanka }

				has_culture_group = culture_group:indo_aryan_group

jarls_conversion.4 = {
	type = character_event
	desc = "EVTDESC_jarls_conversion.4"
	picture = GFX_evt_throne_room
	
	theme = culture_change
	only_rulers = yes

	trigger = {
		#culture = culture:indo_norse
		independent = no
		top_liege = {
			OR = {
				has_title = k_sindh
				has_title = k_punjab
				has_title = k_delhi
				has_title = k_gujarat
				has_title = k_rajputana
				has_title = k_malwa
				has_title = k_kosala
				has_title = k_gondwana
				has_title = k_bengal
				has_title = k_kamarupa
				has_title = k_orissa
				has_title = k_bihar
				has_title = k_maharastra
				has_title = k_lanka
			}
		}
		any_courtier = {
			OR = {
				AND = {
					has_culture_group = culture_group:indo_aryan_group
					NOT = { has_culture = culture:indo_norse }
				}
				set_culture = culture:norse
			}
		}
	}

	option = {
		name = "EVTOPTA_jarls_conversion.4"
		random_courtier = {
			limit = {
				OR = {
					AND = {
						has_culture_group = culture_group:indo_aryan_group
						NOT = { has_culture = culture:indo_norse }
					}
					set_culture = culture:norse
				}
			}
			set_culture = culture:indo_norse
			hidden_tooltip = {
				character_event = { id = jarls_conversion.6 days = 3 }
			}
		}
	}
}
