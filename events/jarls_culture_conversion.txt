namespace = jarls_conversion

# The indo_norse Melting Pot
jarls_conversion.2 = {
	type = character_event
	desc = "EVTDESC_jarls_conversion.2"
	
	theme = culture_change

	trigger = {
		any_county = {
			title_province = {
				OR = {
					geographical_region = world_india_bengal
					geographical_region = world_india_rajastan
				}
			}
			kingdom = title:k_lanka
			culture_group = culture_group:indo_aryan_group
			NOT = { culture = culture:indo_norse }
		}
	}

	option = {
		name = "EVTOPTA_jarls_conversion.2"
		set_culture = culture:indo_norse
		random_county = {
			limit = {
				title_province = {
					OR = {
						geographical_region = world_india_bengal
						geographical_region = world_india_rajastan
					}
				}
				kingdom = title:k_lanka
				culture_group = culture_group:indo_aryan_group
				NOT = { culture = culture:indo_norse }
			}
			set_county_culture	 = culture:indo_norse
		}
	}
}

#Make my court speak indo_norse
jarls_conversion.4 = {
	type = character_event
	desc = "EVTDESC_jarls_conversion.4"
	
	theme = culture_change

	trigger = {
		#culture = culture:indo_norse
		is_independent_ruler = no
		liege = {
			OR = {
				has_title = title:k_sindh
				has_title = title:k_punjab
				has_title = title:k_delhi
				has_title = title:k_gujarat
				has_title = title:k_rajputana
				has_title = title:k_malwa
				has_title = title:k_kosala
				has_title = title:k_gondwana
				has_title = title:k_bengal
				has_title = title:k_kamarupa
				has_title = title:k_orissa
				has_title = title:k_bihar
				has_title = title:k_maharastra
				has_title = title:k_lanka
			}
		}
		any_courtier = {
			OR = {
				AND = {
					has_culture_group = culture_group:indo_aryan_group
					NOT = { has_culture = culture:indo_norse }
				}
				has_culture = culture:norse
			}
		}
	}

	option = {
		name = "EVTOPTA_jarls_conversion.4"
		random_courtier = {
			limit = {
				OR = {
					AND = {
						has_culture_group = culture_group:indo_aryan_group
						NOT = { has_culture = culture:indo_norse }
					}
					has_culture = culture:norse
				}
			}
			set_culture = culture:indo_norse
			hidden_effect = {
				trigger_event = { id = jarls_conversion.6 days = 3 }
			}
		}
	}
}
